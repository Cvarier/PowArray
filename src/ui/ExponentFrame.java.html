574     public void Calculate() {
575 
576         try {
577 
578             x = Integer.parseInt(xInput.getText());
579             n = Integer.parseInt(nInput.getText());
580 
581             if (x < 1 || n < 1) {
582 
583                 throw new NumberFormatException();
584 
585             }
586 
587             resultPane.setText("");
588 
589             StyleConstants.setForeground(attr, Color.decode("#15AD1A"));
590 
591             Runnable r = new Runnable() {
592                 public void run() {
593 
594                     isStarted = true;
595                     isStopped = false;
596                     xInput.setEnabled(false);
597                     nInput.setEnabled(false);
598                     btnCopy.setEnabled(false);
599                     btnSave.setEnabled(false);
600                     btnCalculate.setEnabled(false);
601                     btnBack.setEnabled(false);
602                     btnStop.setEnabled(true);
603                     btnPausePlay.setEnabled(true);
604 
605                     LargePowerCalc.Reinitialize();
606 
607                     // Play Start Sound
608                     if (prefs.getBoolean(soundOn, true)) {
609                         playClip(startSound, 0);
610                     }
611 
612                     try {
613                         eventDoc.insertString(
614                                 eventDoc.getLength(),
615                                 "Calculation started at: "
616                                         + System.getProperty("line.separator")
617                                         + new java.sql.Timestamp(Calendar
618                                                 .getInstance().getTime()
619                                                 .getTime())
620                                         + System.getProperty("line.separator")
621                                         + System.getProperty("line.separator"),
622                                 attr);
623                     } catch (BadLocationException e) {
624                         e.printStackTrace();
625                     }
626 
627                     calcStartTime = System.currentTimeMillis();
628                     LargePowerCalc.exponentiate(x, n);
629 
630                     if (!isStopped) {
631 
632                         calcEndTime = System.currentTimeMillis();
633                         calcRunTime = calcEndTime - calcStartTime;
634                         // Play Finish Sound
635                         if (prefs.getBoolean(soundOn, true)
636                                 && calcRunTime > 580) {
637                             playClip(finishSound, 1);
638                         }
639 
640                         try {
641                             StyleConstants.setForeground(attr,
642                                     Color.decode("#15AD1A"));
643                             eventDoc.insertString(
644                                     eventDoc.getLength(),
645                                     "Calculation finished at: "
646                                             + System.getProperty("line.separator")
647                                             + new java.sql.Timestamp(Calendar
648                                                     .getInstance().getTime()
649                                                     .getTime()), attr);
650                         } catch (BadLocationException e) {
651                             e.printStackTrace();
652                         }
653 
654                     } else {
655 
656                         calcEndTime = System.currentTimeMillis();
657                         calcRunTime = calcEndTime - calcStartTime;
658                         // Play Stop Sound
659                         if (prefs.getBoolean(soundOn, true)
660                                 && calcRunTime > 580) {
661                             playClip(stopSound, 4);
662                         }
663 
664                         StyleConstants.setForeground(attr, Color.RED);
665                         try {
666                             eventDoc.insertString(
667                                     eventDoc.getLength(),
668                                     "Calculation stopped at: "
669                                             + System.getProperty("line.separator")
670                                             + new java.sql.Timestamp(Calendar
671                                                     .getInstance().getTime()
672                                                     .getTime()), attr);
673                         } catch (BadLocationException e) {
674                             e.printStackTrace();
675                         }
676 
677                         isStarted = false;
678                         isStopped = false;
679 
680                     }
681 
682                     btnStop.setEnabled(false);
683                     btnPausePlay.setIcon(pauseIcon);
684                     btnPausePlay.setMargin(new Insets(2, 6, 0, 9));
685                     btnPausePlay.setFont(new Font("Tahoma", Font.BOLD, 16));
686                     btnPausePlay.setText("Pause");
687                     btnPausePlay
688                             .setToolTipText("Pause the calculation so that it can be resumed later.");
689                     isPaused = false;
690                     btnPausePlay.setEnabled(false);
691 
692                     if (prefs.getBoolean(soundOn, true)) {
693 
694                         try {
695                             Thread.sleep(600);
696                         } catch (InterruptedException e) {
697                             e.printStackTrace();
698                         }
699 
700                     }
701 
702                     xInput.setEnabled(true);
703                     nInput.setEnabled(true);
704                     btnCalculate.setEnabled(true);
705                     btnBack.setEnabled(true);
706                     btnCopy.setEnabled(true);
707                     btnSave.setEnabled(true);
708 
709                 }
710             };
711             Thread calculationOngoing = new Thread(r);
712             calculationOngoing.start();
713 
714         } catch (NumberFormatException e1) {
715             // Play Error Sound
716             if (prefs.getBoolean(soundOn, true)) {
717                 playClip(errorSound, 5);
718 
719                 Runnable r = new Runnable() {
720                     public void run() {
721                         btnCalculate.setEnabled(false);
722                         xInput.setEnabled(false);
723                         nInput.setEnabled(false);
724                         btnBack.setEnabled(false);
725                         try {
726                             Thread.sleep(1000);
727                         } catch (InterruptedException e) {
728                             e.printStackTrace();
729                         }
730                         btnCalculate.setEnabled(true);
731                         xInput.setEnabled(true);
732                         nInput.setEnabled(true);
733                         btnBack.setEnabled(true);
734                     }
735                 };

773     private void playClip(String file, int soundType) {
774         try {
775 
776             clip = (Clip) AudioSystem.getLine(new Line.Info(Clip.class));
777 
778             clip.addLineListener(new LineListener() {
779                 @Override
780                 public void update(LineEvent event) {
781                     if (event.getType() == LineEvent.Type.STOP)
782                         clip.close();
783 
784                 }
785             });
786 
787             clip.open(AudioSystem.getAudioInputStream(ExponentFrame.class
788                     .getResource(file)));
789             float volume = 0;
790 
791             // Reduce volume
792             // FloatControl min value is -80.0f
793             if (soundType == 0) {
794                 volume = (float) (-(100 - prefs.getInt(startSoundVol, 60)) * 0.6);
795                 FloatControl gainControl = (FloatControl) clip
796                         .getControl(FloatControl.Type.MASTER_GAIN);
797                 gainControl.setValue(volume);
798                 volume = prefs.getInt(startSoundVol, 60);
799             } else if (soundType == 1) {
800                 volume = (float) (-(100 - prefs.getInt(finishSoundVol, 60)) * 0.6);
801                 FloatControl gainControl = (FloatControl) clip
802                         .getControl(FloatControl.Type.MASTER_GAIN);
803                 gainControl.setValue(volume);
804                 volume = prefs.getInt(startSoundVol, 60);
805             } else if (soundType == 2) {
806                 volume = (float) (-(100 - prefs.getInt(pauseSoundVol, 60)) * 0.6);
807                 FloatControl gainControl = (FloatControl) clip
808                         .getControl(FloatControl.Type.MASTER_GAIN);
809                 gainControl.setValue(volume);
810                 volume = prefs.getInt(startSoundVol, 60);
811             } else if (soundType == 3) {
812                 volume = (float) (-(100 - prefs.getInt(resumeSoundVol, 60)) * 0.6);
813                 FloatControl gainControl = (FloatControl) clip
814                         .getControl(FloatControl.Type.MASTER_GAIN);
815                 gainControl.setValue(volume);
816                 volume = prefs.getInt(startSoundVol, 60);
817             } else if (soundType == 4) {
818                 volume = (float) (-(100 - prefs.getInt(stopSoundVol, 60)) * 0.6);
819                 FloatControl gainControl = (FloatControl) clip
820                         .getControl(FloatControl.Type.MASTER_GAIN);
821                 gainControl.setValue(volume);
822                 volume = prefs.getInt(startSoundVol, 60);
823             } else if (soundType == 5) {
824                 volume = (float) (-(100 - prefs.getInt(errorSoundVol, 60)) * 0.6);
825                 FloatControl gainControl = (FloatControl) clip
826                         .getControl(FloatControl.Type.MASTER_GAIN);
827                 gainControl.setValue(volume);
828                 volume = prefs.getInt(startSoundVol, 60);
829             }
830 
831             if (volume == 0) {
832                 return;
833             }
834 
835             clip.start();
836         } catch (Exception exc) {
837             exc.printStackTrace();
838         }

841     private void chooseFile() {
842 
843         btnSave.setEnabled(false);
844 
845         if (!prefs.getBoolean(alwaysAskForSaveLoc, false)) {
846 
847             String pathName = "Timestamp_"
848                     + new SimpleDateFormat("yyyy MM dd").format(Calendar
849                             .getInstance().getTime()) + "_" + x + "^" + n
850                     + ".txt";
851 
852             if (prefs.get(saveLoc, "").equals("")) {
853 
854                 fileChooser = new MyChooser();
855                 fileChooser.setFileFilter(new FileNameExtensionFilter(".txt",
856                         "txt"));
857                 fileChooser.setFileSelectionMode(JFileChooser.DIRECTORIES_ONLY);
858 
859                 int status = fileChooser.showDialog(null, "Set");
860 
861                 if (status == JFileChooser.APPROVE_OPTION) {
862 
863                     try {
864                         String selectedFileDir = fileChooser.getSelectedFile()
865                                 .getCanonicalPath();
866                         if (Files
867                                 .exists(fileChooser.getSelectedFile().toPath())) {
868                             out = new PrintWriter(fileChooser.getSelectedFile()
869                                     + "\\" + pathName, "UTF-8");
870                             out.println(resultPane.getText());
871                             out.close();
872                             prefs.put(saveLoc, selectedFileDir + "\\");
873                         } else {
874                             prefs.put(saveLoc, "");
875                             btnSave.setEnabled(true);
876                         }
877 
878                     } catch (IOException e1) {
879                         e1.printStackTrace();
880                         btnSave.setEnabled(true);
881                     }
882                 } else if (status == JFileChooser.CANCEL_OPTION) {
883 
884                     btnSave.setEnabled(true);
885 
886                 }
887 
888             } else {
889 
890                 try {
891                     out = new PrintWriter(prefs.get(saveLoc, "") + pathName,
892                             "UTF-8");
893                     out.println(resultPane.getText());
894                     out.close();
895                 } catch (FileNotFoundException e1) {
896                     e1.printStackTrace();
897                     prefs.put(saveLoc, "");
898                     chooseFile();
899                 } catch (UnsupportedEncodingException e2) {
900                     e2.printStackTrace();
901                 }
902 
903             }
904 
905         } else {
906 
907             fileChooser = new MyChooser();
908             fileChooser
909                     .setFileFilter(new FileNameExtensionFilter(".txt", "txt"));
910 
911             int status = fileChooser.showSaveDialog(null);
912 
913             if (status == JFileChooser.APPROVE_OPTION) {
914 
915                 try {
916                     PrintWriter out = new PrintWriter(
917                             fileChooser.getSelectedFile() + ".txt", "UTF-8");
918                     out.println(resultPane.getText());
919                     out.close();
920                 } catch (IOException e1) {
921                     e1.printStackTrace();
922                     btnSave.setEnabled(true);
923                 }
924             } else if (status == JFileChooser.CANCEL_OPTION) {
925 
926                 btnSave.setEnabled(true);
927 
928             }
929 
930         }

505     public void Calculate() {
506 
507         try {
508 
509             x = Integer.parseInt(xInput.getText());
510             resultPane.setText("");
511 
512             StyleConstants.setForeground(attr, Color.decode("#15AD1A"));
513 
514             arraySize = FactorialArraySizeSetter.arraySizeSetter(x);
515 
516             Runnable r = new Runnable() {
517                 public void run() {
518 
519                     isStarted = true;
520                     isStopped = false;
521                     xInput.setEnabled(false);
522                     btnSave.setEnabled(false);
523                     btnCalculate.setEnabled(false);
524                     btnBack.setEnabled(false);
525                     btnStop.setEnabled(true);
526                     btnPausePlay.setEnabled(true);
527                     btnCopy.setEnabled(false);
528 
529                     FactorialCalc.Reinitialize();
530 
531                     // Play Start Sound
532                     if (prefs.getBoolean(soundOn, true)) {
533                         playClip(startSound, 0);
534                     }
535 
536                     try {
537                         eventDoc.insertString(
538                                 eventDoc.getLength(),
539                                 "Calculation started at: "
540                                         + System.getProperty("line.separator")
541                                         + new java.sql.Timestamp(Calendar
542                                                 .getInstance().getTime()
543                                                 .getTime())
544                                         + System.getProperty("line.separator")
545                                         + System.getProperty("line.separator"),
546                                 attr);
547                     } catch (BadLocationException e) {
548                         e.printStackTrace();
549                     }
550 
551                     calcStartTime = System.currentTimeMillis();
552                     FactorialCalc.factorialize(x, arraySize);
553 
554                     if (!isStopped) {
555 
556                         calcEndTime = System.currentTimeMillis();
557                         calcRunTime = calcEndTime - calcStartTime;
558                         // Play Finish Sound
559                         if (prefs.getBoolean(soundOn, true)
560                                 && calcRunTime > 580) {
561                             playClip(finishSound, 1);
562                         }
563 
564                         try {
565                             StyleConstants.setForeground(attr,
566                                     Color.decode("#15AD1A"));
567                             eventDoc.insertString(
568                                     eventDoc.getLength(),
569                                     "Calculation finished at: "
570                                             + System.getProperty("line.separator")
571                                             + new java.sql.Timestamp(Calendar
572                                                     .getInstance().getTime()
573                                                     .getTime()), attr);
574                         } catch (BadLocationException e) {
575                             e.printStackTrace();
576                         }
577                     } else {
578 
579                         calcEndTime = System.currentTimeMillis();
580                         calcRunTime = calcEndTime - calcStartTime;
581                         // Play Stop Sound
582                         if (prefs.getBoolean(soundOn, true)
583                                 && calcRunTime > 580) {
584                             playClip(stopSound, 4);
585                         }
586 
587                         StyleConstants.setForeground(attr, Color.RED);
588                         try {
589                             eventDoc.insertString(
590                                     eventDoc.getLength(),
591                                     "Calculation stopped at: "
592                                             + System.getProperty("line.separator")
593                                             + new java.sql.Timestamp(Calendar
594                                                     .getInstance().getTime()
595                                                     .getTime()), attr);
596                         } catch (BadLocationException e) {
597                             e.printStackTrace();
598                         }
599 
600                         isStarted = false;
601                         isStopped = false;
602 
603                     }
604 
605                     btnStop.setEnabled(false);
606                     btnPausePlay.setIcon(pauseIcon);
607                     btnPausePlay.setMargin(new Insets(2, 6, 0, 9));
608                     btnPausePlay.setFont(new Font("Tahoma", Font.BOLD, 16));
609                     btnPausePlay.setText("Pause");
610                     btnPausePlay
611                             .setToolTipText("Pause the calculation so that it can be resumed later.");
612                     isPaused = false;
613                     btnPausePlay.setEnabled(false);
614 
615                     if (prefs.getBoolean(soundOn, true)) {
616 
617                         try {
618                             Thread.sleep(600);
619                         } catch (InterruptedException e) {
620                             e.printStackTrace();
621                         }
622 
623                     }
624 
625                     xInput.setEnabled(true);
626                     btnCalculate.setEnabled(true);
627                     btnBack.setEnabled(true);
628                     btnCopy.setEnabled(true);
629                     btnSave.setEnabled(true);
630 
631                 }
632             };
633             Thread calculationOngoing = new Thread(r);
634             calculationOngoing.start();
635 
636         } catch (NumberFormatException e1) {
637             // Play Error Sound
638             if (prefs.getBoolean(soundOn, true)) {
639                 playClip(errorSound, 5);
640 
641                 Runnable r = new Runnable() {
642                     public void run() {
643                         btnCalculate.setEnabled(false);
644                         xInput.setEnabled(false);
645                         btnBack.setEnabled(false);
646                         try {
647                             Thread.sleep(1000);
648                         } catch (InterruptedException e) {
649                             e.printStackTrace();
650                         }
651                         btnCalculate.setEnabled(true);
652                         xInput.setEnabled(true);
653                         btnBack.setEnabled(true);
654                     }
655                 };
656                 Thread timer = new Thread(r);
657                 timer.start();
658             }
659             resultPane.setText("");
660             try {
661                 StyleConstants.setForeground(attr, Color.RED);
662                 eventDoc.insertString(
663                         eventDoc.getLength(),
664                         "Please enter a positive integer less than or equal to 2000000, "
665                                 + "and then either press enter or click on calculate.",
666                         attr);
667             } catch (BadLocationException e2) {
668                 e2.printStackTrace();
669             }
670         }

692     private void playClip(String file, int soundType) {
693         try {
694 
695             clip = (Clip) AudioSystem.getLine(new Line.Info(Clip.class));
696 
697             clip.addLineListener(new LineListener() {
698                 @Override
699                 public void update(LineEvent event) {
700                     if (event.getType() == LineEvent.Type.STOP)
701                         clip.close();
702 
703                 }
704             });
705 
706             clip.open(AudioSystem.getAudioInputStream(ExponentFrame.class
707                     .getResource(file)));
708             float volume = 0;
709 
710             // Reduce volume
711             // FloatControl min value is -80.0f
712             if (soundType == 0) {
713                 volume = (float) (-(100 - prefs.getInt(startSoundVol, 60)) * 0.6);
714                 FloatControl gainControl = (FloatControl) clip
715                         .getControl(FloatControl.Type.MASTER_GAIN);
716                 gainControl.setValue(volume);
717                 volume = prefs.getInt(startSoundVol, 60);
718             } else if (soundType == 1) {
719                 volume = (float) (-(100 - prefs.getInt(finishSoundVol, 60)) * 0.6);
720                 FloatControl gainControl = (FloatControl) clip
721                         .getControl(FloatControl.Type.MASTER_GAIN);
722                 gainControl.setValue(volume);
723                 volume = prefs.getInt(startSoundVol, 60);
724             } else if (soundType == 2) {
725                 volume = (float) (-(100 - prefs.getInt(pauseSoundVol, 60)) * 0.6);
726                 FloatControl gainControl = (FloatControl) clip
727                         .getControl(FloatControl.Type.MASTER_GAIN);
728                 gainControl.setValue(volume);
729                 volume = prefs.getInt(startSoundVol, 60);
730             } else if (soundType == 3) {
731                 volume = (float) (-(100 - prefs.getInt(resumeSoundVol, 60)) * 0.6);
732                 FloatControl gainControl = (FloatControl) clip
733                         .getControl(FloatControl.Type.MASTER_GAIN);
734                 gainControl.setValue(volume);
735                 volume = prefs.getInt(startSoundVol, 60);
736             } else if (soundType == 4) {
737                 volume = (float) (-(100 - prefs.getInt(stopSoundVol, 60)) * 0.6);
738                 FloatControl gainControl = (FloatControl) clip
739                         .getControl(FloatControl.Type.MASTER_GAIN);
740                 gainControl.setValue(volume);
741                 volume = prefs.getInt(startSoundVol, 60);
742             } else if (soundType == 5) {
743                 volume = (float) (-(100 - prefs.getInt(errorSoundVol, 60)) * 0.6);
744                 FloatControl gainControl = (FloatControl) clip
745                         .getControl(FloatControl.Type.MASTER_GAIN);
746                 gainControl.setValue(volume);
747                 volume = prefs.getInt(startSoundVol, 60);
748             }
749 
750             if (volume == 0) {
751                 return;
752             }
753 
754             clip.start();
755         } catch (Exception exc) {
756             exc.printStackTrace();
757         }

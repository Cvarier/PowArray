519     private void playClip(String file, int soundType) {
520         try {
521 
522             clip = (Clip) AudioSystem.getLine(new Line.Info(Clip.class));
523 
524             clip.addLineListener(new LineListener() {
525                 @Override
526                 public void update(LineEvent event) {
527                     if (event.getType() == LineEvent.Type.STOP)
528                         clip.close();
529 
530                 }
531             });
532 
533             clip.open(AudioSystem.getAudioInputStream(ExponentFrame.class
534                     .getResource(file)));
535             float volume = 0;
536 
537             // Reduce volume
538             // FloatControl min value is -80.0f
539             if (soundType == 0) {
540                 volume = (float) (-(100 - prefs.getInt(startSoundVol, 60)) * 0.6);
541                 FloatControl gainControl = (FloatControl) clip
542                         .getControl(FloatControl.Type.MASTER_GAIN);
543                 gainControl.setValue(volume);
544                 volume = prefs.getInt(startSoundVol, 60);
545             } else if (soundType == 1) {
546                 volume = (float) (-(100 - prefs.getInt(finishSoundVol, 60)) * 0.6);
547                 FloatControl gainControl = (FloatControl) clip
548                         .getControl(FloatControl.Type.MASTER_GAIN);
549                 gainControl.setValue(volume);
550                 volume = prefs.getInt(startSoundVol, 60);
551             } else if (soundType == 2) {
552                 volume = (float) (-(100 - prefs.getInt(pauseSoundVol, 60)) * 0.6);
553                 FloatControl gainControl = (FloatControl) clip
554                         .getControl(FloatControl.Type.MASTER_GAIN);
555                 gainControl.setValue(volume);
556                 volume = prefs.getInt(startSoundVol, 60);
557             } else if (soundType == 3) {
558                 volume = (float) (-(100 - prefs.getInt(resumeSoundVol, 60)) * 0.6);
559                 FloatControl gainControl = (FloatControl) clip
560                         .getControl(FloatControl.Type.MASTER_GAIN);
561                 gainControl.setValue(volume);
562                 volume = prefs.getInt(startSoundVol, 60);
563             } else if (soundType == 4) {
564                 volume = (float) (-(100 - prefs.getInt(stopSoundVol, 60)) * 0.6);
565                 FloatControl gainControl = (FloatControl) clip
566                         .getControl(FloatControl.Type.MASTER_GAIN);
567                 gainControl.setValue(volume);
568                 volume = prefs.getInt(startSoundVol, 60);
569             } else if (soundType == 5) {
570                 volume = (float) (-(100 - prefs.getInt(errorSoundVol, 60)) * 0.6);
571                 FloatControl gainControl = (FloatControl) clip
572                         .getControl(FloatControl.Type.MASTER_GAIN);
573                 gainControl.setValue(volume);
574                 volume = prefs.getInt(startSoundVol, 60);
575             }
576 
577             if (volume == 0) {
578                 return;
579             }
580 
581             clip.start();
582         } catch (Exception exc) {
583             exc.printStackTrace();
584         }
